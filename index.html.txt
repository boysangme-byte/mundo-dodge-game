<body>
  <h2>🏹 문도 도끼 피하기</h2>
  <canvas id="game" width="480" height="640"></canvas>
  <div id="hud">점수: <span id="score">0</span></div>

  <script>
    const canvas = document.getElementById("game");
    const ctx = canvas.getContext("2d");

    // 플레이어
    const player = { x: 220, y: 580, w: 40, h: 40, speed: 5 };

    // 도끼(투사체) 배열
    const axes = [];
    let score = 0;
    let gameOver = false;

    // 키 입력 처리
    const keys = {};
    window.addEventListener("keydown", e => keys[e.key] = true);
    window.addEventListener("keyup", e => keys[e.key] = false);

    function spawnAxe() {
      const x = Math.random() * (canvas.width - 20);
      axes.push({ x, y: -20, w: 20, h: 20, speed: 3 + Math.random() * 3 });
    }

    function update() {
      if (gameOver) return;

      // 플레이어 이동
      if (keys["ArrowLeft"] && player.x > 0) player.x -= player.speed;
      if (keys["ArrowRight"] && player.x + player.w < canvas.width) player.x += player.speed;
      if (keys["ArrowUp"] && player.y > 0) player.y -= player.speed;
      if (keys["ArrowDown"] && player.y + player.h < canvas.height) player.y += player.speed;

      // 도끼 이동
      for (let axe of axes) {
        axe.y += axe.speed;
        // 충돌 판정
        if (
          axe.x < player.x + player.w &&
          axe.x + axe.w > player.x &&
          axe.y < player.y + player.h &&
          axe.y + axe.h > player.y
        ) {
          gameOver = true;
          alert("게임 오버! 최종 점수: " + score);
          document.location.reload();
        }
      }

      // 화면 밖 도끼 제거 + 점수 증가
      for (let i = axes.length - 1; i >= 0; i--) {
        if (axes[i].y > canvas.height) {
          axes.splice(i, 1);
          score++;
          document.getElementById("score").textContent = score;
        }
      }
    }

    function draw() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      // 플레이어 (파란 사각형)
      ctx.fillStyle = "cyan";
      ctx.fillRect(player.x, player.y, player.w, player.h);

      // 도끼 (보라색 사각형 = 문도 도끼)
      ctx.fillStyle = "violet";
      for (let axe of axes) {
        ctx.fillRect(axe.x, axe.y, axe.w, axe.h);
      }
    }

    function loop() {
      update();
      draw();
      requestAnimationFrame(loop);
    }

    // 일정 주기로 도끼 생성
    setInterval(spawnAxe, 800);
    loop();
  </script>
